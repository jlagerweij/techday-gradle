Script


#!/bin/sh
case $1 in
    classes)
        mkdir out
        for file in *.java
        do
           javac $file -d out/
        done
        ;;
    clean)
        rm -rf out
        ;;
    esac

Make


JFLAGS = -g
JC = javac
.SUFFIXES: .java .class
.java.class:
        $(JC) $(JFLAGS) $*.java

CLASSES = \
        Foo.java \
        Blah.java \
        Library.java \
        Main.java

default: classes

classes: $(CLASSES:.java=.class)

clean:
        $(RM) *.class
