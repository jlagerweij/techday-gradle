Script


#!/bin/sh
case $1 in
    classes)
        mkdir out
        for file in *.java
        do
           javac $file -d out/
        done
        ;;
    clean)
        rm -rf out
        ;;
    esac

Make


JFLAGS = -g
JC = javac
.SUFFIXES: .java .class
.java.class:
        $(JC) $(JFLAGS) $*.java

CLASSES = \
        Foo.java \
        Blah.java \
        Library.java \
        Main.java

default: classes

classes: $(CLASSES:.java=.class)

clean:
        $(RM) *.class

apply plugin: 'java'

    maven { url "http://nexus.insight.sbp.avisi.nl:8081/nexus/content/groups/avisi-nexus" }

    compile group: 'com.moreapps', name: 'api', version: '5.3.1'
    testCompile group: 'junit', name: 'junit', version: '4.+'
}

    systemProperties 'property': 'value'

    repositories {
